39 inline std::array<double, 2> toCartesian(const std::array<double, 2> &WGS84Reference, const std::array<double, 2> &WGS84Position) {
40 #ifndef M_PI
41  constexpr double M_PI = 3.141592653589793;
42 #endif
43  constexpr double DEG_TO_RAD{M_PI / 180.0};
44  constexpr double HALF_PI{M_PI / 2.0};
45  constexpr double EPSILON10{1.0e-10};
46  constexpr double EPSILON12{1.0e-12};
47 
48  constexpr double EQUATOR_RADIUS{6378137.0};
49  constexpr double FLATTENING{1.0 / 298.257223563};
50  constexpr double SQUARED_ECCENTRICITY{2.0 * FLATTENING - FLATTENING * FLATTENING};
51  constexpr double SQUARE_ROOT_ONE_MINUS_ECCENTRICITY{0.996647189335};
52  constexpr double POLE_RADIUS{EQUATOR_RADIUS * SQUARE_ROOT_ONE_MINUS_ECCENTRICITY};
53 
54  constexpr double C00{1.0};
55  constexpr double C02{0.25};
56  constexpr double C04{0.046875};
57  constexpr double C06{0.01953125};
58  constexpr double C08{0.01068115234375};
59  constexpr double C22{0.75};
60  constexpr double C44{0.46875};
61  constexpr double C46{0.01302083333333333333};
62  constexpr double C48{0.00712076822916666666};
63  constexpr double C66{0.36458333333333333333};
64  constexpr double C68{0.00569661458333333333};
65  constexpr double C88{0.3076171875};
66 
67  constexpr double R0{C00 - SQUARED_ECCENTRICITY * (C02 + SQUARED_ECCENTRICITY * (C04 + SQUARED_ECCENTRICITY * (C06 + SQUARED_ECCENTRICITY * C08)))};
68  constexpr double R1{SQUARED_ECCENTRICITY * (C22 - SQUARED_ECCENTRICITY * (C04 + SQUARED_ECCENTRICITY * (C06 + SQUARED_ECCENTRICITY * C08)))};
69  constexpr double R2T{SQUARED_ECCENTRICITY * SQUARED_ECCENTRICITY};
70  constexpr double R2{R2T * (C44 - SQUARED_ECCENTRICITY * (C46 + SQUARED_ECCENTRICITY * C48))};
71  constexpr double R3T{R2T * SQUARED_ECCENTRICITY};
72  constexpr double R3{R3T * (C66 - SQUARED_ECCENTRICITY * C68)};
73  constexpr double R4{R3T * SQUARED_ECCENTRICITY * C88};
74 
75  auto mlfn = [&](const double &lat) {
76  const double sin_phi{std::sin(lat)};
77  const double cos_phi{std::cos(lat) * sin_phi};
78  const double squared_sin_phi = sin_phi * sin_phi;
79  return (R0 * lat - cos_phi * (R1 + squared_sin_phi * (R2 + squared_sin_phi * (R3 + squared_sin_phi * R4))));
80  };
81 
82  const double ML0{mlfn(WGS84Reference[0] * DEG_TO_RAD)};
83 
84  auto msfn = [&](const double &sinPhi, const double &cosPhi, const double &es) { return (cosPhi / std::sqrt(1.0 - es * sinPhi * sinPhi)); };
85 
86  auto project = [&](double lat, double lon) {
87  std::array<double, 2> retVal{lon, -1.0 * ML0};
88  if (!(std::abs(lat) < EPSILON10)) {
89  const double ms{(std::abs(std::sin(lat)) > EPSILON10) ? msfn(std::sin(lat), std::cos(lat), SQUARED_ECCENTRICITY) / std::sin(lat) : 0.0};
90  retVal[0] = ms * std::sin(lon *= std::sin(lat));
91  retVal[1] = (mlfn(lat) - ML0) + ms * (1.0 - std::cos(lon));
92  }
93  return retVal;
94  };
95 
96  auto fwd = [&](double lat, double lon) {
97  const double D = std::abs(lat) - HALF_PI;
98  if ((D > EPSILON12) || (std::abs(lon) > 10.0)) {
99  return std::array<double, 2>{0.0, 0.0};
100  }
101  if (std::abs(D) < EPSILON12) {
102  lat = (lat < 0.0) ? -1.0 * HALF_PI : HALF_PI;
103  }
104  lon -= WGS84Reference[1] * DEG_TO_RAD;
105  const auto projectedRetVal{project(lat, lon)};
106  return std::array<double, 2>{EQUATOR_RADIUS * projectedRetVal[0], EQUATOR_RADIUS * projectedRetVal[1]};
107  };
108 
109  return fwd(WGS84Position[0] * DEG_TO_RAD, WGS84Position[1] * DEG_TO_RAD);
110 }

117 inline std::array<double, 2> fromCartesian(const std::array<double, 2> &WGS84Reference, const std::array<double, 2> &CartesianPosition) {
118  constexpr double EPSILON10{1.0e-2};
119  constexpr double incLon{1e-5};
120  const int32_t signLon{(CartesianPosition[0] < 0) ? -1 : 1};
121  constexpr double incLat{incLon};
122  const int32_t signLat{(CartesianPosition[1] < 0) ? -1 : 1};
123 
124  std::array<double, 2> approximateWGS84Position{WGS84Reference};
125  std::array<double, 2> cartesianResult{toCartesian(WGS84Reference, approximateWGS84Position)};
126 
127  double dPrev{std::numeric_limits<double>::max()};
128  double d{std::abs(CartesianPosition[1] - cartesianResult[1])};
129  while ((d < dPrev) && (d > EPSILON10)) {
130  approximateWGS84Position[0] = approximateWGS84Position[0] + signLat * incLat;
131  cartesianResult = toCartesian(WGS84Reference, approximateWGS84Position);
132  dPrev = d;
133  d = std::abs(CartesianPosition[1] - cartesianResult[1]);
134  }
135 
136  dPrev = std::numeric_limits<double>::max();
137  d = std::abs(CartesianPosition[0] - cartesianResult[0]);
138  while ((d < dPrev) && (d > EPSILON10)) {
139  approximateWGS84Position[1] = approximateWGS84Position[1] + signLon * incLon;
140  cartesianResult = toCartesian(WGS84Reference, approximateWGS84Position);
141  dPrev = d;
142  d = std::abs(CartesianPosition[0] - cartesianResult[0]);
143  }
144 
145  return approximateWGS84Position;
146 }
